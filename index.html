function generateEverything() { const name = document.getElementById('brandName').value; if(!name) return alert('يرجى كتابة الاسم أولاً!'); const btn = document.getElementById('genBtn'); const loading = document.getElementById('loadingArea'); const result = document.getElementById('resultArea'); const img = document.getElementById('brandImage'); const fallback = document.getElementById('textLogoFallback'); const placeholderText = document.getElementById('logoTextPlaceholder'); btn.disabled = true; loading.classList.remove('hidden'); result.classList.add('hidden'); img.classList.add('hidden'); fallback.classList.add('hidden'); const seed = Math.floor(Math.random() * 5000); // جلب صورة هوية بصرية img.src = `https://picsum.photos/seed/${seed}/800/800`; img.onload = () => { loading.classList.add('hidden'); result.classList.remove('hidden'); img.classList.remove('hidden'); btn.disabled = false; // تحديث النصوص document.getElementById('sloganText').innerHTML = `
${name} | مستقبل الإبداع

🎁 ملصقك الرقمي الجاهز:

${name}
`; }; img.onerror = () => { loading.classList.add('hidden'); result.classList.remove('hidden'); fallback.classList.remove('hidden'); placeholderText.innerText = name; btn.disabled = false; }; } // دالة بسيطة لمحاكاة حفظ الملصق function downloadSticker() { alert('سيتم تصدير الملصق كصورة PNG عالية الجودة قريباً!'); }
