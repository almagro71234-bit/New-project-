async function generateWithAI() { const name = document.getElementById('brandInput').value; if(!name) return alert('يرجى إدخال اسم المشروع'); const btn = document.getElementById('aiBtn'); const img = document.getElementById('aiImage'); const resultDiv = document.getElementById('aiResult'); const loader = document.getElementById('loadingStatus'); btn.disabled = true; btn.innerText = 'جاري البحث في المحركات...'; resultDiv.classList.remove('hidden'); loader.classList.remove('hidden'); img.classList.add('hidden'); const seed = Math.floor(Math.random() * 99999); // قائمة المحركات (أكثر من محرك لضمان العمل) const engines = [ `https://image.pollinations.ai/prompt/professional-logo-design-for-${name}-vector-style?seed=${seed}&nologo=true`, // المحرك 1: Pollinations `https://api.qrserver.com/v1/create-qr-code/?size=800x800&data=${name}`, // المحرك 2: محرك فني (تجريبي) `https://picsum.photos/seed/${seed}/800/800` // المحرك 3: محرك صور عالي الجودة (احتياطي) ]; let currentEngine = 0; function tryNextEngine() { if (currentEngine < engines.length) { console.log(`محاولة المحرك رقم: ${currentEngine + 1}`); img.src = engines[currentEngine]; currentEngine++; } else { // في حال فشل الجميع alert('جميع المحركات مشغولة حالياً، يرجى المحاولة بعد دقيقة'); btn.disabled = false; loader.classList.add('hidden'); } } // ابدأ بالمحرك الأول tryNextEngine(); img.onload = () => { loader.classList.add('hidden'); img.classList.remove('hidden'); btn.disabled = false; btn.innerText = 'توليد بالذكاء الاصطناعي 🚀'; }; // إذا فشل المحرك الحالي، انتقل للي بعده تلقائياً img.onerror = () => { tryNextEngine(); }; }
